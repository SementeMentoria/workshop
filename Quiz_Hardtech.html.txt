            document.querySelectorAll('.screen').forEach(screen => {

        function updateProgress() {
            const percentage = ((currentQuestion + 1) / 12) * 100;
            document.querySelector('.progress-fill').style.width = percentage + '%';
            document.getElementById('progress-text').textContent = `${currentQuestion + 1}/12`;
        }

        function saveCurrentAnswer() {
            const selected = document.querySelector(`input[name="q${currentQuestion}"]:checked`);
            if (selected) {
                answers[currentQuestion] = parseInt(selected.value);
            }
        }

        function showQuestion(idx) {
            currentQuestion = idx;
            document.getElementById('question-title').textContent = `Pergunta ${idx + 1}: ${questions[idx].question}`;
            const optionsDiv = document.getElementById('options');
            optionsDiv.innerHTML = '';
            questions[idx].options.forEach((option, i) => {
                const label = document.createElement('label');
                label.className = 'option';
                label.innerHTML = `<input type="radio" name="q${idx}" value="${i}" ${answers[idx] === i ? 'checked' : ''}> ${option}`;
                optionsDiv.appendChild(label);
            });
            updateProgress();
            document.getElementById('prev-btn').disabled = idx === 0;
            document.getElementById('next-btn').textContent = idx === 11 ? 'Finalizar Quiz' : 'Próximo';
        }

        function getScores() {
            const points = { A: 0, B: 0, C: 0, D: 0, E: 0 };
            const tiePoints = { A: 0, B: 0, C: 0, D: 0, E: 0 };
            for (let i = 0; i < 12; i++) {
                if (answers[i] === -1) return null;
                const letter = String.fromCharCode(65 + answers[i]);
                const pt = tripleQuestions.includes(i) ? 3 : 1;
                points[letter] += pt;
                if (tripleQuestions.includes(i)) {
                    tiePoints[letter] += pt;
                }
            }
            return { points, tiePoints };
        }

        function findWinner(scores) {
            const { points, tiePoints } = scores;
            const maxPoints = Math.max(...Object.values(points));
            let candidates = Object.keys(points).filter(key => points[key] === maxPoints);
            if (candidates.length === 1) {
                return candidates[0];
            }
            // Tiebreaker: max tiePoints among candidates
            let maxTie = -1;
            let winner = candidates[0];
            for (let cand of candidates) {
                if (tiePoints[cand] > maxTie) {
                    maxTie = tiePoints[cand];
                    winner = cand;
                }
            }
            return winner;
        }

            showScreen('result');
